    pageEncoding="UTF-8"%>
<!DOCTYPE html>
<html>
<head>
    <!-- 문서의 문자 인코딩을 UTF-8로 설정 -->
    <meta charset="UTF-8">
    <title>Navigation Header</title>
</head>
<body>
    <!-- 네비게이션 메뉴를 포함하는 테이블 -->
    <!-- width를 90%로 설정하여 화면 크기에 반응형으로 동작 -->
    <table border="1" style="width: 90%">
        <tr>
            <!-- 모든 메뉴 항목을 중앙 정렬 -->
            <td align="center">
                <%-- 로그인 상태에 따른 조건부 메뉴 렌더링 --%>
                <%-- session의 UserId 속성을 검사하여 로그인 상태 확인 --%>
                <%
                if(session.getAttribute("UserId") == null){ 
                    // 로그인하지 않은 상태: 로그인 링크 표시
                %>
                    <a href="../Process/LoginProcess.jsp">로그인</a>
                <% } else { 
                    // 로그인한 상태: 로그아웃 링크 표시
                %>
                    <a href="../Process/LogoutProcess.jsp">로그아웃</a>
                < }
                %>
                
                <%-- 메뉴 항목 간 간격을 위한 공백 --%>
                &nbsp;&nbsp;&nbsp;
                
                <%-- 페이징이 없는 게시판으로 이동하는 링크 --%>
                <a href="../Main/BoardList.jsp">게시판(페이징X)</a>
                
                <%-- 메뉴 항목 간 간격을 위한 공백 --%>
                &nbsp;&nbsp;&nbsp;
                
                <%-- 페이징이 있는 게시판으로 이동하는 링크 --%>
                <a href="../Main/BoardPageHandlerList.jsp">게시판(페이징O)</a>
            </td>
        </tr>
    </table>
</body>

이 JSP 파일은 웹 애플리케이션의 상단 네비게이션 메뉴를 제공하며, 사용자의 로그인 상태에 따라 동적으로 메뉴를 변경하는 헤더 컴포넌트로 사용됩니다.
