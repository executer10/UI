//JSFunction.java
package DataBaseUtils;    // 데이터베이스 유틸리티 관련 패키지

import javax.servlet.jsp.JspWriter;    // JSP 페이지에 출력하기 위한 출력 스트림 클래스

/**
 * JSP 페이지에서 자주 사용되는 자바스크립트 함수들을 모아둔 유틸리티 클래스
 * 주로 알림창 표시와 페이지 이동에 관한 기능을 제공
 */
public class JSFunction {
    /**
     * 알림창을 띄운 후 지정된 URL로 페이지를 이동시키는 메서드
     * 
     * @param msg 알림창에 표시할 메시지 내용
     * @param url 메시지 표시 후 이동할 페이지의 URL
     * @param out JSP 페이지의 출력 스트림 객체
     */
    public static void alertLocation(String msg, String url, JspWriter out) {
        try {
            // 자바스크립트 코드를 문자열로 생성
            String script = ""  // 빈 문자열로 시작
                    + "<script>"  // 자바스크립트 코드 시작
                    + "    alert('" + msg + "');"  // alert() 함수로 메시지 표시
                    + "    location.href='"+ url +"';"  // location.href로 페이지 이동
                    + "</script>";  // 자바스크립트 코드 끝
            
            // 만들어진 자바스크립트 코드를 JSP 페이지에 출력
            out.println(script);
        }
        catch (Exception e) {
            // 예외 발생시 처리할 코드
            // 현재는 빈 catch 블록으로 되어있지만, 
            // 실제 운영 환경에서는 로그를 남기거나 적절한 예외 처리를 해주는 것이 좋습니다.
        }
    }
    
    /**
     * 알림창을 띄운 후 브라우저의 이전 페이지로 돌아가는 메서드
     * 
     * @param msg 알림창에 표시할 메시지 내용
     * @param out JSP 페이지의 출력 스트림 객체
     */
    public static void alertBack(String msg, JspWriter out) {
        try {
            // 자바스크립트 코드를 문자열로 생성
            String script = ""  // 빈 문자열로 시작
                    + "<script>"  // 자바스크립트 코드 시작
                    + "    alert('" + msg + "');"  // alert() 함수로 메시지 표시
                    + "    history.back();"  // history.back()으로 이전 페이지로 이동
                    + "</script>";  // 자바스크립트 코드 끝
            
            // 만들어진 자바스크립트 코드를 JSP 페이지에 출력
            out.println(script);
        } 
        catch (Exception e) {
            // 예외 발생시 처리할 코드
            // 현재는 빈 catch 블록으로 되어있지만,
            // 실제 운영 환경에서는 로그를 남기거나 적절한 예외 처리를 해주는 것이 좋습니다.
        }
    }
}

이 클래스는 JSP 페이지에서 자주 사용되는 자바스크립트 기능들을 자바 코드로 쉽게 생성할 수 있도록 도와주는 유틸리티 클래스입니다. 특히 사용자에게 메시지를 보여주고 페이지를 이동시키는 기능을 자주 사용할 때 유용합니다.
